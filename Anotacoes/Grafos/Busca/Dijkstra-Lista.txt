#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const ll INF = 1e18;

void dijkstraList(int s, const vector<vector<pair<int,ll>>>& adj, vector<ll>& dist) {
    int n = adj.size();
    dist.assign(n, INF);
    dist[s] = 0;

    priority_queue<pair<ll,int>, vector<pair<ll,int>>, greater<>> pq;
    pq.push({0, s});

    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue;

        for (auto [v, w] : adj[u]) {
            if (dist[v] > dist[u] + w) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
}
