ll binary_search_first(vector<ll> &vet, ll value) {
    ll l = 0, r = (ll)vet.size() - 1;
    ll ans = -1;
    while (l <= r) {
        ll mid = l + (r - l) / 2;
        if (vet[mid] == value) {
            ans = mid;       // encontrou
            r = mid - 1;     // busca a primeira ocorrÃªncia
        } 
        else if (vet[mid] > value) {
            r = mid - 1;
        } 
        else {
            l = mid + 1;
        }
    }
    return ans;
}
